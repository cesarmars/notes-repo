# Terminal Commands 

## Home directory
tilde = ~ = /User/Username

Example:
- ~/code/...

## Moving folders or files to a different directory
move = mv

Example:
- mv ~/Downloads/file.txt  ~/61c/

## See current directory
print working directory = pwd

Example: 
- pwd -> "/Users/cesarmartinez/code"

## Print text and interact with files
echoes = echo

Examples:
- echo hello world (or) echo "hello world"
- echo hello world > file.txt

## Copy a file
copy = cp

Example:
- cp file.txt ~/Desktop/

## Copy a folder to another folder
copy recurvsive = cp -r

Example:
-cp -r myfolder ~/Documents/

## Remove folder
If the folder is empty -> rmdir filename

If the folder is non empty -> rm -r foldername

## Change a file name 
move = mv

Example:
- mv file.txt distination.txt

## Changing directory (backwards)
change directory = cd

Example:
- cd ~/Downloads

## Prompt

% or $ are prompt symbols

## Run program
current directoy = .

dirctory seperator = /

- ./ = run the program in the current directory
